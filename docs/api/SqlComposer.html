module Select

SqlComposer.Select

Generate composable SQL Select queries and query fragments.

Here is a basic example:

let sql =
  SqlComposer.Select.(
    select
    |. field("*")
    |. from("test")
    |. where("AND test.foo = ?")
    |. to_sql
  );
Js.log2("SQL: ", sql);
